date_pattern = re.compile(r"\d{1,2}/\d{1,2}/\d{2}")

def reformat_dates(col_name: str) -> str:
    """Reformat dates to the day/month/year format""" 
    try:
        return date_pattern.sub(datetime.strptime(col_name, "%m/%d/%y").strftime("%d/%m/%Y"), col_name, count=1)
    except ValueError:
        return col_name

def frame_args(duration):
    """Helper function for producing the frames for the map animation"""
    return {
        "frame": {"duration": duration},
        "mode": "immediate",
        "fromcurrent": True,
        "transition": {"duration": duration, "easing": "linear"},
    }

# Create plotly subplot
fig = make_subplots(rows=2, cols=1, specs=[[{"type": "scattergeo"}], [{"type": "xy"}]], row_heights=[0.8, 0.2])

# Set up colorbar tick values, ranging from 1 to the highest number of confirmed cases for any country thus far
max_country_confirmed = cases_by_date[dates_list[-1]]["confirmed"].max()

# To account for the significant variance in number of cases, make the scale logarithmic
high_tick = np.log1p(max_country_confirmed)
low_tick = np.log1p(1)
log_tick_values = np.geomspace(low_tick, high_tick, num=6)

# However, we want the labels on the scale to be the actual number of cases (i.e. not log(n_cases))
visual_tick_values = np.expm1(log_tick_values).astype(int)

# Explicitly set max cbar value, otherwise it might be max - 1 due to a rounding error
visual_tick_values[-1] = max_country_confirmed  
visual_tick_values = [f"{val:,}" for val in visual_tick_values]

# Generate line chart data as list of tuples: [(confirmed, deaths), ...]
cases_deaths_totals = [
    (df.filter(like="confirmed").astype("uint32").agg("sum")[0], df.filter(like="deaths").astype("uint32").agg("sum")[0]) for df in cases_by_date.values()
]

confirmed_totals = [daily_total[0] for daily_total in cases_deaths_totals]

# Reload data
confirmed_df = pd.read_csv(CONFIRMED_URL).rename(columns=renamed_columns_map).rename(columns=reformat_dates).fillna(method="bfill", axis=1)
deaths_df = pd.read_csv(DEATHS_URL).rename(columns=renamed_columns_map).rename(columns=reformat_dates).fillna(method="bfill", axis=1)

fig = go.Figure()

geo_data_cols = ["country", "location", "latitude", "longitude"]
geo_data_df = confirmed_df[geo_data_cols]
dates_list = (
    confirmed_df.filter(regex=r"(\d{2}/\d{2}/\d{4})", axis=1)
    .columns
    .to_list()
)

# Create a dictionary of dataframes by date, where each df holds the daily counts of cases and deaths per country
cases_by_date = {}
for date in dates_list:
    # Get a pd.Series of all cases for the current day
    confirmed_day_df = confirmed_df.filter(like=date, axis=1).rename(columns=lambda col: "confirmed").astype("uint32")
    
    # Get a pd.Series of all deaths for the current day
    deaths_day_df = deaths_df.filter(like=date, axis=1).rename(columns=lambda col: "deaths").astype("uint32")
    
    cases_df = confirmed_day_df.join(deaths_day_df)  # combine the cases and deaths dfs
    cases_df = geo_data_df.join(cases_df)  # add in the geographical data
    cases_df = cases_df[cases_df["confirmed"] > 0]  # get rid of any rows where there were no cases
    
    cases_by_date[date] = cases_df

annotation_text_template = "<b>Worldwide Totals</b>" \
                           "<br>{date}<br><br>" \
                           "Confirmed cases: {confirmed:,d}<br>" \
                           "Deaths: {deaths:,d}<br>" \
                           "Mortality rate: {mortality_rate:.1%}"
annotation_dict = {
    "x": 0.15,
    "y": 0.35,
    "width": 175,
    "height": 110,
    "showarrow": False,
    "text": "",
    "valign": "middle",
    "visible": False,
    "bordercolor": "black",
}

steps = []
for i, data in enumerate(fig.data):
    step = {
        "method": "update",
        "args": [
            {"visible": [False] * len(fig.data)},
            {"annotations": [dict(annotation_dict) for _ in range(len(fig.data))]},
        ],
        "label": dates_list[i],
    }

    # Toggle the i'th trace and annotation box to visible
    step["args"][0]["visible"][i] = True
    step["args"][1]["annotations"][i]["visible"] = True

    df = cases_by_date[dates_list[i]]
    confirmed = df["confirmed"].sum()
    deaths = df["deaths"].sum()
    mortality_rate = deaths / confirmed
    step["args"][1]["annotations"][i]["text"] = annotation_text_template.format(
        date=dates_list[i],
        confirmed=confirmed,
        deaths=deaths,
        mortality_rate=mortality_rate,
    )

    steps.append(step)

sliders = [
